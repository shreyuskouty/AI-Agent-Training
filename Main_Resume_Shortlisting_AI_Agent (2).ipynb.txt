        "# CELL 4 ▸ Groq client & LLM wrapper\n",
        "def get_groq_client(api_key):\n",
        "    return Groq(api_key=api_key)\n",
        "\n",
        "def llm_call(client, system_prompt, user_prompt, max_tokens=2000):\n",
        "    for attempt in range(3):\n",
        "        try:\n",
        "            resp = client.chat.completions.create(\n",
        "                model='llama-3.1-8b-instant',\n",
        "                messages=[\n",
        "                    {'role':'system','content':system_prompt},\n",
        "                    {'role':'user',  'content':user_prompt}\n",
        "                ],\n",
        "                max_tokens=max_tokens,\n",
        "                temperature=0.2,\n",
        "            )\n",
        "            return resp.choices[0].message.content.strip()\n",
        "        except Exception as e:\n",
        "            if attempt == 2: raise e\n",
        "            time.sleep(2 ** attempt)\n",
        "    return ''\n",
        "print('✅ Groq LLM helper ready — llama-3.1-8b-instant')"

        "def extract_text_from_file(file_path, file_name):\n",
        "    ext = os.path.splitext(file_name)[1].lower()\n",
        "    try:\n",
        "        if ext == '.pdf':\n",
        "            pages = []\n",
        "            with open(file_path, 'rb') as f:\n",
        "                reader = PyPDF2.PdfReader(f)\n",
        "                for page in reader.pages:\n",
        "                    t = page.extract_text()\n",
        "                    if t: pages.append(t)\n",
        "            return '\\n'.join(pages)\n",
        "        elif ext in ('.docx', '.doc'):\n",
        "            doc = docx.Document(file_path)\n",
        "            return '\\n'.join(p.text for p in doc.paragraphs)\n",
        "        else:\n",
        "            with open(file_path, 'r', errors='ignore') as f:\n",
        "                return f.read()\n",
        "    except Exception as e:\n",
        "        return f'[Error: {e}]'\n",
        "print('✅ File extractor ready — PDF, DOCX, TXT')"

        "# CELL 7 ▸ Build LangGraph pipeline\n",
        "def build_graph(groq_api_key):\n",
        "    client = get_groq_client(groq_api_key)\n",
        "    def perception(s):       return node_perception(s, client)\n",
        "    def memory_retrieval(s): return node_memory_retrieval(s, client)\n",
        "    def reasoning(s):        return node_reasoning_planning(s, client)\n",
        "    def tool_sel(s):         return node_tool_selection(s)\n",
        "    def action(s):           return node_action_execution(s, client)\n",
        "    def llm_resp(s):         return node_llm_response(s, client)\n",
        "    def final_out(s):        return node_final_output(s)\n",
        "\n",
        "    g = StateGraph(AgentState)\n",
        "    g.add_node('perception',       perception)\n",
        "    g.add_node('memory_retrieval', memory_retrieval)\n",
        "    g.add_node('reasoning',        reasoning)\n",
        "    g.add_node('tool_selection',   tool_sel)\n",
        "    g.add_node('action_execution', action)\n",
        "    g.add_node('llm_response',     llm_resp)\n",
        "    g.add_node('final_output',     final_out)\n",
        "    g.set_entry_point('perception')\n",
        "    g.add_edge('perception',       'memory_retrieval')\n",
        "    g.add_edge('memory_retrieval', 'reasoning')\n",
        "    g.add_edge('reasoning',        'tool_selection')\n",
        "    g.add_edge('tool_selection',   'action_execution')\n",
        "    g.add_edge('action_execution', 'llm_response')\n",
        "    g.add_edge('llm_response',     'final_output')\n",
        "    g.add_edge('final_output',     END)\n",
